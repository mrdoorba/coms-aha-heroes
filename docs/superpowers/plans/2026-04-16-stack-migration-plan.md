# AHA Heroes Stack Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate the AHA Heroes frontend from React/TanStack to SvelteKit while keeping the Elysia backend, using an Inverted Monolith architecture with Bun workspaces.

**Architecture:** Elysia is the primary HTTP server. It handles `/api/*` routes directly and delegates all other requests to SvelteKit for SSR page rendering. Both run in a single Bun process, sharing the database connection pool and session cookies. The codebase is organized as a Bun workspace with three packages: `server`, `web`, and `shared`.

**Tech Stack:** Bun, Elysia 1.3, SvelteKit 2 + Svelte 5 (runes), BetterAuth, Drizzle ORM, TypeBox, Eden Treaty, shadcn-svelte, LayerChart, Superforms, Paraglide, @vite-pwa/sveltekit

**Design Spec:** `docs/superpowers/specs/2026-04-16-stack-migration-design.md`

- [ ] **Step 1: Write the generation script**

```typescript
// packages/shared/scripts/generate-schemas.ts
import { createInsertSchema, createSelectSchema } from 'drizzle-typebox'
import { Type } from '@sinclair/typebox'
import * as schema from '../src/db/schema'
import { writeFileSync } from 'fs'
import { resolve } from 'path'

// Tables to generate schemas for
const tables = {
  users: schema.users,
  userEmails: schema.userEmails,
  branches: schema.branches,
  teams: schema.teams,
  pointCategories: schema.pointCategories,
  achievementPoints: schema.achievementPoints,
  challenges: schema.challenges,
  appeals: schema.appeals,
  comments: schema.comments,
  rewards: schema.rewards,
  redemptions: schema.redemptions,
  notifications: schema.notifications,
  pointSummaries: schema.pointSummaries,
  auditLogs: schema.auditLogs,
  sheetSyncJobs: schema.sheetSyncJobs,
  systemSettings: schema.systemSettings,
} as const

type TableEntries = typeof tables

// Generate insert and select schemas for each table
const lines: string[] = [
  '// AUTO-GENERATED by scripts/generate-schemas.ts — do not edit manually',
  '// Run: bun run generate:schemas',
  '',
  "import { createInsertSchema, createSelectSchema } from 'drizzle-typebox'",
  "import { Type } from '@sinclair/typebox'",
  "import * as schema from '../db/schema'",
  '',
]

for (const [name, _table] of Object.entries(tables)) {
  const pascalName = name.charAt(0).toUpperCase() + name.slice(1)
  lines.push(`// --- ${pascalName} ---`)
  lines.push(`export const insert${pascalName}Schema = createInsertSchema(schema.${name})`)
  lines.push(`export const select${pascalName}Schema = createSelectSchema(schema.${name})`)
  lines.push('')
}

// Add safe response schemas (omitting sensitive fields)
lines.push('// --- Safe Response Schemas (sensitive fields omitted) ---')
lines.push("export const safeUserSchema = Type.Omit(selectUsersSchema, ['mustChangePassword'])")
lines.push('')

const outPath = resolve(import.meta.dir, '../src/schemas/index.ts')
writeFileSync(outPath, lines.join('\n'))
console.log(`[generate-schemas] wrote ${outPath}`)
```

// packages/web/src/hooks.server.ts
import type { Handle } from '@sveltejs/kit'
import { auth } from '@coms/server/auth'
import { db } from '@coms/shared/db'
import { users, userEmails } from '@coms/shared/db/schema'
import { eq } from 'drizzle-orm'
import type { AuthUser } from '@coms/shared/types'

export const handle: Handle = async ({ event, resolve }) => {
  // Read session from BetterAuth cookie (same-process, no HTTP call)
  const session = await auth.api.getSession({
    headers: event.request.headers,
  })

  if (!session) {
    event.locals.user = null
    event.locals.session = null
    return resolve(event)
  }

  // Look up app user by primary email, then secondary emails
  let [appUser] = await db
    .select({
      id: users.id,
      email: users.email,
      name: users.name,
      role: users.role,
      branchId: users.branchId,
      teamId: users.teamId,
      canSubmitPoints: users.canSubmitPoints,
      mustChangePassword: users.mustChangePassword,
    })
    .from(users)
    .where(eq(users.email, session.user.email))
    .limit(1)

  if (!appUser) {
    const [secondary] = await db
      .select({ userId: userEmails.userId })
      .from(userEmails)
      .where(eq(userEmails.email, session.user.email))
      .limit(1)

    if (secondary) {
      ;[appUser] = await db
        .select({
          id: users.id,
          email: users.email,
          name: users.name,
          role: users.role,
          branchId: users.branchId,
          teamId: users.teamId,
          canSubmitPoints: users.canSubmitPoints,
          mustChangePassword: users.mustChangePassword,
        })
        .from(users)
        .where(eq(users.id, secondary.userId))
        .limit(1)
    }
  }

  event.locals.user = (appUser as AuthUser) ?? null
  event.locals.session = session.session
    ? {
        id: session.session.id,
        userId: session.session.userId,
        expiresAt: new Date(session.session.expiresAt),
      }
    : null

  return resolve(event)
}

    }
  }
